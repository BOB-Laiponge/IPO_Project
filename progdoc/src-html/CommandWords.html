003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This class holds an enumeration table of all command words known to the game.
008 * It is used to recognise commands as they are typed in.
009 *
010 * @author  Michael Kolling and David J. Barnes + D.Bureau + PY.Pitiot
011 * @version 2008.03.30 + 2019.09.25 + 2020.04.21
012 */
013public class CommandWords
014{
015    // a constant array that will hold all valid command words
016    private final String[] aValidCommands = 
017    {"go", "help","quit","look","eat","back","test", "take", "drop", "items", "use", "load", "alea"};
018
019    /**
020     * Constructor - initialise the command words.
021     */
022    public CommandWords()
023    {
024        //rien Ã  faire 
025        
026    } // CommandWords()
027
028    /**
029     * Check whether a given String is a valid command word. 
030     * @return true if a given string is a valid command,
031     * false if it isn't.
032     */
033    public boolean isCommand( final String pString )
034    {
035        for ( int i=0; i<this.aValidCommands.length; i++ ) {
036            if ( this.aValidCommands[i].equals( pString ) )
037                return true;
038        } // for
039        // if we get here, the string was not found in the commands
040        return false;
041    } // isCommand()
042    
043    /**
044     * Affiche les commandes valides 
045     */
046    public String getCommandList()
047    {
048        String vCommandString = "";
049        for(String command : this.aValidCommands) {
050            vCommandString += (command + " ");
051        }
052        return vCommandString;
053    }//showAll
054    
055} // CommandWords
