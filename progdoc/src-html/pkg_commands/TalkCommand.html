014public class TalkCommand extends Command
015{
016    /**
017     * Constructeur pour les objets de classe TalkCommand.
018     */
019    public TalkCommand()
020    {
021    }
022
023    /**
024     * Commande "talk" : Permet parler Ã  un PNJ.
025     */
026    public void execute(final Player pPlayer, final GameEngine pGE, final UserInterface pGUI)
027    {
028        if (hasSecondWord())
029        {
030            if (pPlayer.getCurrentRoom().hasPNJ(getSecondWord()))
031            {
032                Character vPNJ = pPlayer.getCurrentRoom().getPNJ(getSecondWord());
033                
034                if (vPNJ.getName().equals("Ragnar") || vPNJ.getName().equals("Commandant")){
035                    pGE.summonBombInWM();
036                }
037                
038                pGUI.print(vPNJ.getName() + " : ");
039                pGUI.println(vPNJ.talk());
040                if (vPNJ.getName().equals("Conducteur")){
041                    pGE.goTo(pGE.getRoom("UnionShip"));
042                }
043                return;
044            }
045            pGUI.println("Cette personne n'est pas ici.");
046        }
047        else
048        {
049            pGUI.println("A qui voulez vous parler ?");
050        }
051    }
052}
