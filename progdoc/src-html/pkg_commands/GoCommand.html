027    public void execute(final Player pPlayer, final GameEngine pGE, final UserInterface pGUI)
028    {
029        // Si un seul mot, on retourne "go where ?"
030        if (!hasSecondWord()) {
031            // if there is no second word, we don't know where to go...
032            pGUI.println( "Go where?" );
033            return;
034        }
035
036        // On cherche la prochaine piÃ¨ce
037        String vDirection = this.getSecondWord();
038
039        Room vNextRoom;
040
041        if (pPlayer.getCurrentRoom().isTransporterRoom())
042        {
043            if (pGE.getTestMode() && pGE.getAleaRoom() != null)
044            {
045                vNextRoom = pGE.getAleaRoom();
046            }
047            else
048            {
049                vNextRoom = ((TransporterRoom)(pPlayer.getCurrentRoom())).getExit(vDirection);
050            }
051        }
052        else
053        {
054            vNextRoom = pPlayer.getCurrentRoomExit(vDirection);
055        }
056
057        // On effectue ou pas le changement de lieu
058        if ( vNextRoom == null )
059            pGUI.println( "There is no door!" );
060        else {
061
062
063            pPlayer.decreaseTurnsLeft();
064
065            if (pPlayer.TestTurnsLeft())
066            {
067                pGUI.println("GAME OVER : La bataille au dessus de Saand s'est achevÃ©e par l'annihilation de la planÃ¨te.\nVous Ãªtes mort !");
068                pGE.endGame();

073
074        }
075    }
