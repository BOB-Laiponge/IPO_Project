200    /**
201     * Given a command, process (that is: execute) the command.
202     * If this command ends the game, true is returned, otherwise false is
203     * returned.
204     */
205    public void interpretCommand( final String pCommandLine ) 
206    {
207        this.aGui.println( "> " + pCommandLine );
208        Command vCommand = this.aParser.getCommand( pCommandLine );
209
210        if ( vCommand.isUnknown() ) {
211            this.aGui.println( "I don't know what you mean..." );
212            return;
213        }
214
215        String vCommandWord = vCommand.getCommandWord();
216        if ( vCommandWord.equals( "help" ) )
217            this.printHelp();
218        else if ( vCommandWord.equals( "go" ) )
219            this.goRoom(vCommand);
220        else if ( vCommandWord.equals( "eat" ) )
221            this.eat(vCommand);
222        else if ( vCommandWord.equals( "look" ) )
223            this.look(vCommand);
224        else if ( vCommandWord.equals( "quit" ) ) 
225            this.quit(vCommand);
226        else if ( vCommandWord.equals( "back" ) ) 
227            this.back(vCommand);
228        else if (vCommandWord.equals("test"))
229            this.test(vCommand);
230        else if (vCommandWord.equals("take"))
231            this.take(vCommand);
232        else if (vCommandWord.equals("drop"))
233            this.drop(vCommand);
234        else if (vCommandWord.equals("items"))
235            this.items(vCommand);
236        else if (vCommandWord.equals("use"))
237            this.use(vCommand);
238        else if (vCommandWord.equals("load"))
239            this.load(vCommand);
240    }

286    /**
287     * GÃ¨re le changement de lieu
288     * 
289     * @params La commande entrÃ©e par le joueur
290     */
291    private void goRoom(final Command pCommand)
292    {
293        // Si un seul mot, on retourne "go where ?"
294        if ( ! pCommand.hasSecondWord() ) {
295            // if there is no second word, we don't know where to go...
296            this.aGui.println( "Go where?" );
297            return;
298        }
299
300        // On cherche la prochaine piÃ¨ce
301        String vDirection = pCommand.getSecondWord();
302
303        Room vNextRoom = this.aPlayer.getCurrentRoomExit(vDirection);
304
305        // On effectue ou pas le changement de lieu
306        if ( vNextRoom == null )
307            this.aGui.println( "There is no door!" );
308        else {
309            this.goTo(vNextRoom);
310        }
311
312    }//goRoom()
